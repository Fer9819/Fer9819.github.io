  • Función para oprtimizar ————————————————
calc_idw <- function(x){
  print(x)
  pts <- lts[[x]] 
  a <- unique(pts$Profund)
  coordinates(pts) <- ~ x + y
  crs(pts) <- crs(fnca) 
  idw.p <- gstat::idw(DA ~1,pts,grd)
  idw.p <- raster(idw.p)
  idw.p <- raster::mask(idw.p,fnca)
  dir.create("DA")
  writeRaster(idw.p,paste0("DA/idw_DA",a, ".tif"), overwrite= TRUE)
}
ord <- c(1:length(lts))
rsl <- map(.x= ord, .f= calc_idw)
